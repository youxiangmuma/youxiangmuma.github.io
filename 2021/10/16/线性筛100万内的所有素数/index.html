//线性筛100万内的所有素数
void get_prime(){
for(int i = 2; i < N; i++){
if(!st[i]) prime[cnt++] = i;
for(int j = 0; prime[j] <= N / i; j++){
st[prime[j] * i] = true;
if(i % prime[j] == 0) break;
}
}
}
